    Example H: Using SDI-12 in Slave Mode.

    Example H: Using SDI-12 in Slave Mode

    Copyright

    Stroud Water Research Center

    This example is published under the BSD-3 license.

    Date

    2016

    Author

    D. Wasielewski

    Example sketch demonstrating how to implement an arduino as a slave on an SDI-12 bus. This may be used, for example, as a middleman between an I2C sensor and an SDI-12 datalogger.

    Note that an SDI-12 slave must respond to M! or C! with the number of values it will report and the max time until these values will be available. This example uses 9 values available in 21 s, but references to these numbers and the output array size and datatype should be changed for your specific application.

    D. Wasielewski, 2016 Builds upon work started by: https://github.com/jrzondagh/AgriApps-SDI-12-Arduino-Sensor https://github.com/Jorge-Mendes/Agro-Shield/tree/master/SDI-12ArduinoSensor

    Suggested improvements:

    • Get away from memory-hungry arduino String objects in favor of char buffers
    • Make an int variable for the "number of values to report" instead of the hard-coded 9s interspersed throughout the code

    57void parseSdi12Cmd(String command, String* dValues) {
    58 /* Ingests a command from an SDI-12 master, sends the applicable response, and
    59 * (when applicable) sets a flag to initiate a measurement
    60 */
    61
    62 // First char of command is always either (a) the address of the device being
    63 // probed OR (b) a '?' for address query.
    64 // Do nothing if this command is addressed to a different device
    65 if (command.charAt(0) != sensorAddress && command.charAt(0) != '?') { return; }
    66
    67 // If execution reaches this point, the slave should respond with something in
    68 // the form: <address><responseStr><Carriage Return><Line Feed>
    69 // The following if-switch-case block determines what to put into <responseStr>,
    70 // and the full response will be constructed afterward. For '?!' (address query)
    71 // or 'a!' (acknowledge active) commands, responseStr is blank so section is skipped
    72 String responseStr = "";
    73 if (command.length() > 1) {
    74 switch (command.charAt(1)) {
    75 case 'I':
    76 // Identify command
    77 // Slave should respond with ID message: 2-char SDI-12 version + 8-char
    78 // company name + 6-char sensor model + 3-char sensor version + 0-13 char S/N
    79 responseStr = "13COMPNAME0000011.0001"; // Substitute proper ID String here
    80 break;
    81 case 'C':
    82 // Initiate concurrent measurement command
    83 // Slave should immediately respond with: "tttnn":
    84 // 3-digit (seconds until measurement is available) +
    85 // 2-digit (number of values that will be available)
    86 // Slave should also start a measurment and relinquish control of the data line
    87 responseStr =
    88 "02109"; // 9 values ready in 21 sec; Substitue sensor-specific values here
    89 // It is not preferred for the actual measurement to occur in this subfunction,
    90 // because doing to would hold the main program hostage until the measurement
    91 // is complete. Instead, we'll just set a flag and handle the measurement
    92 // elsewhere.
    93 state = INITIATE_CONCURRENT;
    94 break;
    95 // NOTE: "aC1...9!" commands may be added by duplicating this case and adding
    96 // additional states to the state flag
    97 case 'M':
    98 // Initiate measurement command
    99 // Slave should immediately respond with: "tttnn":
    100 // 3-digit (seconds until measurement is available) +
    101 // 1-digit (number of values that will be available)
    102 // Slave should also start a measurment but may keep control of the data line
    103 // until advertised time elapsed OR measurement is complete and service request
    104 // sent
    105 responseStr =
    106 "0219"; // 9 values ready in 21 sec; Substitue sensor-specific values here
    107 // It is not preferred for the actual measurement to occur in this subfunction,
    108 // because doing to would hold the main program hostage until the measurement is
    109 // complete. Instead, we'll just set a flag and handle the measurement
    110 // elsewhere. It is preferred though not required that the slave send a service
    111 // request upon completion of the measurement. This should be handled in the
    112 // main loop().
    113 state = INITIATE_MEASUREMENT;
    114 break;
    115 // NOTE: "aM1...9!" commands may be added by duplicating this case and adding
    116 // additional states to the state flag
    117
    118 case 'D':
    119 // Send data command
    120 // Slave should respond with a String of values
    121 // Values to be returned must be split into Strings of 35 characters or fewer
    122 // (75 or fewer for concurrent). The number following "D" in the SDI-12 command
    123 // specifies which String to send
    124 responseStr = dValues[(int)command.charAt(2) - 48];
    125 break;
    126 case 'A':
    127 // Change address command
    128 // Slave should respond with blank message (just the [new] address + <CR> +
    129 // <LF>)
    130 sensorAddress = command.charAt(2);
    131 break;
    132 default:
    133 // Mostly for debugging; send back UNKN if unexpected command received
    134 responseStr = "UNKN";
    135 break;
    136 }
    137 }
    138
    139 // Issue the response specified in the switch-case structure above.
    140 String fullResponse = String(sensorAddress) + responseStr + "\r\n";
    141 slaveSDI12.sendResponse(fullResponse);
    142}

    144void formatOutputSDI(float* measurementValues, String* dValues, unsigned int maxChar) {
    145 /* Ingests an array of floats and produces Strings in SDI-12 output format */
    146
    147 dValues[0] = "";
    148 int j = 0;
    149
    150 // upper limit on i should be number of elements in measurementValues
    151 for (int i = 0; i < 9; i++) {
    152 // Read float value "i" as a String with 6 deceimal digits
    153 // (NOTE: SDI-12 specifies max of 7 digits per value; we can only use 6
    154 // decimal place precision if integer part is one digit)
    155 String valStr = String(measurementValues[i], 6);
    156 // Explictly add implied + sign if non-negative
    157 if (valStr.charAt(0) != '-') { valStr = '+' + valStr; }
    158 // Append dValues[j] if it will not exceed 35 (aM!) or 75 (aC!) characters
    159 if (dValues[j].length() + valStr.length() < maxChar) {
    160 dValues[j] += valStr;
    161 }
    162 // Start a new dValues "line" if appending would exceed 35/75 characters
    163 else {
    164 dValues[++j] = valStr;
    165 }
    166 }
    167
    168 // Fill rest of dValues with blank strings
    169 while (j < 9) { dValues[++j] = ""; }
    170}

    178void loop() {
    179 static float measurementValues[9]; // 9 floats to hold simulated sensor data
    180 static String
    181 dValues[10]; // 10 String objects to hold the responses to aD0!-aD9! commands
    182 static String commandReceived = ""; // String object to hold the incoming command
    183
    184
    185 // If a byte is available, an SDI message is queued up. Read in the entire message
    186 // before proceding. It may be more robust to add a single character per loop()
    187 // iteration to a static char buffer; however, the SDI-12 spec requires a precise
    188 // response time, and this method is invariant to the remaining loop() contents.
    189 int avail = slaveSDI12.available();
    190 if (avail < 0) {
    191 slaveSDI12.clearBuffer();
    192 } // Buffer is full; clear
    193 else if (avail > 0) {
    194 for (int a = 0; a < avail; a++) {
    195 char charReceived = slaveSDI12.read();
    196 // Character '!' indicates the end of an SDI-12 command; if the current
    197 // character is '!', stop listening and respond to the command
    198 if (charReceived == '!') {
    199 state = PROCESS_COMMAND;
    200 // Command string is completed; do something with it
    201 parseSdi12Cmd(commandReceived, dValues);
    202 // Clear command string to reset for next command
    203 commandReceived = "";
    204 // '!' should be the last available character anyway, but exit the "for" loop
    205 // just in case there are any stray characters
    206 slaveSDI12.clearBuffer();
    207 // eliminate the chance of getting anything else after the '!'
    208 slaveSDI12.forceHold();
    209 break;
    210 }
    211 // If the current character is anything but '!', it is part of the command
    212 // string. Append the commandReceived String object.
    213 else {
    214 // Append command string with new character
    215 commandReceived += String(charReceived);
    216 }
    217 }
    218 }
    219
    220 // For aM! and aC! commands, parseSdi12Cmd will modify "state" to indicate that
    221 // a measurement should be taken
    222 switch (state) {
    223 case WAIT:
    224 {
    225 break;
    226 }
    227 case INITIATE_CONCURRENT:
    228 {
    229 // Do whatever the sensor is supposed to do here
    230 // For this example, we will just create arbitrary "simulated" sensor data
    231 // NOTE: Your application might have a different data type (e.g. int) and
    232 // number of values to report!
    233 pollSensor(measurementValues);
    234 // Populate the "dValues" String array with the values in SDI-12 format
    235 formatOutputSDI(measurementValues, dValues, 75);
    236 state = WAIT;
    237 slaveSDI12.forceListen(); // sets SDI-12 pin as input to prepare for incoming
    238 // message AGAIN
    239 break;
    240 }
    241 case INITIATE_MEASUREMENT:
    242 {
    243 // Do whatever the sensor is supposed to do here
    244 // For this example, we will just create arbitrary "simulated" sensor data
    245 // NOTE: Your application might have a different data type (e.g. int) and
    246 // number of values to report!
    247 pollSensor(measurementValues);
    248 // Populate the "dValues" String array with the values in SDI-12 format
    249 formatOutputSDI(measurementValues, dValues, 35);
    250 // For aM!, Send "service request" (<address><CR><LF>) when data is ready
    251 String fullResponse = String(sensorAddress) + "\r\n";
    252 slaveSDI12.sendResponse(fullResponse);
    253 state = WAIT;
    254 slaveSDI12.forceListen(); // sets SDI-12 pin as input to prepare for incoming
    255 // message AGAIN
    256 break;
    257 }
    258 case PROCESS_COMMAND:
    259 {
    260 state = WAIT;
    261 slaveSDI12.forceListen();
    262 break;
    263 }
    264 }
    265}
